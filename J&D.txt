  //   1. Debounced timer ~1000ms after the most recent console.* call
  //   2. visibilitychange -> hidden (tab switched away; user is not watching)
  //   3. beforeunload (page closing; sendBeacon survives unload, fetch may not)
  // The legacy postMessage 'flush-console-logs' handler is kept intact for
  // backwards compatibility with the legacy CF agent's getBrowserLogs tool
  // during the parity-flag rollout.
  //
  // Note: .browser-log lives in /home/user/ (sibling of WEBSITE_DIR), so
  // writes don't trigger Vite's file watcher / full-reload. The SSR-error
  // skip below is now belt-and-suspenders — there are no useful runtime
  // logs from Vite's static error template anyway, so suppressing capture
  // there is still cheap insurance.
  if (!__viteSsrErrorSig) (function() {
    var buffer = [];
    var flushing = false;
    var debounceTimer = null;
    var DEBOUNCE_MS = 1000;
    var levels = ['log', 'warn', 'error', 'info'];
    var originals = {};
    levels.forEach(function(level) {
      originals[level] = console[level];
      console[level] = function() {
        try {
          originals[level].apply(console, arguments);
          var msg = Array.prototype.slice.call(arguments).map(function(a) {
            try { return typeof a === 'object' ? JSON.stringify(a) : String(a); }
            catch(e) { return String(a); }
          }).join(' ');
          var ts = new Date().toISOString();
          buffer.push('[' + ts + '] [' + level.toUpperCase() + '] ' + msg);
          if (buffer.length > 200) buffer = buffer.slice(-200);
          // Schedule a debounced flush. Each new console call resets the
          // timer, so a burst of N logs in <DEBOUNCE_MS coalesces into a
          // single POST after the burst settles.
          if (debounceTimer) clearTimeout(debounceTimer);
          debounceTimer = setTimeout(flush, DEBOUNCE_MS);
        } catch(e) {}
      };
    });
    function flush() {
      if (debounceTimer) { clearTimeout(debounceTimer); debounceTimer = null; }
      if (!buffer.length || flushing) return;
      flushing = true;
      var batch = buffer.splice(0, buffer.length);
      try {
        fetch('/__console', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ logs: batch })
        }).catch(function(){}).finally(function(){ flushing = false; });
      } catch(e) { flushing = false; }
    }
    function flushSync() {
      // Used during page unload — fetch() is unreliable when the document is
      // tearing down, but sendBeacon is designed for exactly this case. The
      // browser queues the POST and lets it finish even after the page dies.
      if (debounceTimer) { clearTimeout(debounceTimer); debounceTimer = null; }
      if (!buffer.length) return;
      var batch = buffer.splice(0, buffer.length);
      try {
        if (navigator && typeof navigator.sendBeacon === 'function') {
          var blob = new Blob([JSON.stringify({ logs: batch })], { type: 'application/json' });
          navigator.sendBeacon('/__console', blob);
        } else {
          // Older browser fallback: fire-and-forget fetch with keepalive.
          fetch('/__console', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ logs: batch }),
            keepalive: true
          }).catch(function(){});
        }
      } catch(e) {}
    }
    window.addEventListener('message', function(event) {
      if (event.data && event.data.type === 'flush-console-logs') flush();
    });
    document.addEventListener('visibilitychange', function() {
      if (document.visibilityState === 'hidden') flush();
    });
    window.addEventListener('beforeunload', flushSync);
    window.addEventListener('pagehide', flushSync);
  })();

  // Iframe navigation tracking
  if (window.parent !== window) {
    function notifyNavigation() {
      window.parent.postMessage({
        type: 'navigation',
        path: window.location.pathname + window.location.search + window.location.hash
      }, '*');
    }
    notifyNavigation();
    window.addEventListener('popstate', notifyNavigation);
    window.addEventListener('hashchange', notifyNavigation);
    var originalPushState = history.pushState;
    var originalReplaceState = history.replaceState;
    history.pushState = function() {
      originalPushState.apply(this, arguments);
      notifyNavigation();
    };
    history.replaceState = function() {
      originalReplaceState.apply(this, arguments);
      notifyNavigation();
    };
    window.addEventListener('message', function(event) {
      if (event.data && (event.data.type === 'navigate' || event.data.type === 'navigate-to-path')) {
        var path = event.data.path;
        if (path) {
          window.dispatchEvent(new CustomEvent('landingsite:navigate', { detail: { path: path } }));
          if (window.location.pathname + window.location.search + window.location.hash !== path) {
            // Use originalReplaceState to update the URL without triggering our override's
            // notifyNavigation (the popstate event below will handle the single notification).
            // replaceState avoids adding a duplicate entry to the iframe's history stack.
            originalReplaceState.call(history, { landingsite: true }, '', path);
            // Dispatch popstate so React Router detects the location change and re-renders.
            // pushState/replaceState alone do NOT trigger popstate, which is what React Router listens for.
            window.dispatchEvent(new PopStateEvent('popstate', { state: { landingsite: true } }));
          }
        }
      }
    });
  }

  // Element picker: inert by default. Activates when the parent posts
  // 'enter-select' from an allowlisted origin. Tears down on 'cancel-select',
  // on Escape inside the iframe, or after a successful pick. No listeners on
  // document, no DOM rendered, no cursor change while inactive.
  if (window.parent !== window) {
    var PICKER_ALLOWED_PARENT_ORIGINS = ["https://app.landingsite.ai"];
    // Last-resort sanity cap before postMessage. The parent owns the
    // authoritative size limits (200KB outerHTML, 220KB total per Unit 5);
    // this only catches pathological payloads to keep the channel sane.
    var PICKER_MAX_MESSAGE_BYTES = 500 * 1024;

    var pickActive = false;
    var pickerOverlay = null;
    var pickerHint = null;
    var pickerLearnedParentOrigin = null;

    function pickerCapByBytes(text, maxBytes) {
      var s = String(text == null ? '' : text).trim();
      if (!s) return '';
      try {
        var enc = new TextEncoder();
        var bytes = enc.encode(s);
        if (bytes.length <= maxBytes) return s;
        // Truncate by bytes, then decode with fatal=false so we drop any
        // partial multibyte sequence at the end safely.
        var truncated = bytes.slice(0, maxBytes);
        try {
          return new TextDecoder('utf-8', { fatal: false }).decode(truncated);
        } catch (e) {
          return s.slice(0, maxBytes);
        }
      } catch (e) {
        return s.slice(0, maxBytes);
      }
    }

    // Extract a human-readable text preview from a picked element.
    //
    // el.textContent concatenates source text with no separators, which
    // turns adjacent inline siblings into a runon (e.g. a list of city
    // span elements with no whitespace between them renders as
    // "BoulderLouisvilleLafayette..."). Strip tags with a space substitute
    // and collapse runs of whitespace so adjacent siblings get a separator.
    // Same logic the parent uses in extractPickerSummary so the composer
    // chip and the sent-message indicator agree.
    function pickerSmartText(el) {
      var html;
      try { html = el.outerHTML || ''; } catch (err) { return ''; }
      return html
        .replace(/<[^>]*>/g, ' ')
        .replace(/\s+/g, ' ')
        .trim();
    }

    function pickerEnsureOverlay() {
      if (pickerOverlay && pickerHint) return;
      pickerOverlay = document.createElement('div');
      pickerOverlay.className = 'ls-picker-overlay';
      pickerOverlay.style.top = '-9999px';
      pickerOverlay.style.left = '-9999px';
      pickerOverlay.style.width = '0px';
      pickerOverlay.style.height = '0px';
      pickerHint = document.createElement('div');
      pickerHint.className = 'ls-picker-hint';
      pickerHint.textContent = 'Click an element \u00b7 Esc to cancel';
      var host = document.body || document.documentElement;
      host.appendChild(pickerOverlay);
      host.appendChild(pickerHint);
    }

    function pickerPositionOverlay(target) {
      if (!pickerOverlay || !target || !target.getBoundingClientRect) return;
      var r = target.getBoundingClientRect();
      // Hide overlay for zero-area or off-screen targets (e.g. <html>, <body>
      // when elementFromPoint returns them at the edge).
      if (r.width <= 0 || r.height <= 0) {
        pickerOverlay.style.top = '-9999px';
        pickerOverlay.style.left = '-9999px';
        pickerOverlay.style.width = '0px';
        pickerOverlay.style.height = '0px';
        return;
      }
      pickerOverlay.style.top = r.top + 'px';
      pickerOverlay.style.left = r.left + 'px';
      pickerOverlay.style.width = r.width + 'px';
      pickerOverlay.style.height = r.height + 'px';
    }

    function pickerOnMouseMove(e) {
      // Overlay has pointer-events: none, so elementFromPoint returns the
      // real underlying element rather than the overlay.
      var target = document.elementFromPoint(e.clientX, e.clientY);
      if (!target) return;
      pickerPositionOverlay(target);
    }

    function pickerOnClick(e) {
      e.preventDefault();
      e.stopPropagation();
      // Always use event.target — overlay is pointer-events: none so this is
      // the actual element under the cursor, not a stale hover cache.
      var el = e.target;
      if (!el || !el.tagName) {
        pickerDeactivate();
        return;
      }
      var tag = String(el.tagName).toLowerCase().slice(0, 64);
      var outerHTML = '';
      try { outerHTML = el.outerHTML || ''; } catch (err) { outerHTML = ''; }
      var textPreview = pickerCapByBytes(pickerSmartText(el), 90);
      var payload = {
        type: 'selected',
        outerHTML: outerHTML,
        tag: tag,
        textPreview: textPreview,
        path: window.location.pathname,
      };
      // Sanity cap. Parent enforces the authoritative byte limits.
      try {
        var serializedSize = (outerHTML.length || 0) + (textPreview.length || 0) + 256;
        if (serializedSize > PICKER_MAX_MESSAGE_BYTES) {
          if (typeof console !== 'undefined' && console.warn) {
            console.warn('[ls-picker] dropping pick: serialized message exceeds ~500KB');
          }
          pickerDeactivate();
          return;
        }
      } catch (err) {}
      var targetOrigin = pickerLearnedParentOrigin;
      if (targetOrigin) {
        try { window.parent.postMessage(payload, targetOrigin); } catch (err) {}
      }
      pickerDeactivate();
    }

    function pickerOnKeyDown(e) {
      if (e.key === 'Escape' || e.keyCode === 27) {
        e.preventDefault();
        e.stopPropagation();
        var targetOrigin = pickerLearnedParentOrigin;
        if (targetOrigin) {
          try {
            window.parent.postMessage({ type: 'cancel-select-request' }, targetOrigin);
          } catch (err) {}
        }
        pickerDeactivate();
      }
    }

    function pickerActivate() {
      if (pickActive) return;
      pickActive = true;
      pickerEnsureOverlay();
      try { document.documentElement.style.cursor = 'crosshair'; } catch (e) {}
      // Capture phase + non-passive so we can preventDefault on click and
      // keep customer click handlers from running on the picked element.
      document.addEventListener('mousemove', pickerOnMouseMove, true);
      document.addEventListener('click', pickerOnClick, { capture: true, passive: false });
      document.addEventListener('keydown', pickerOnKeyDown, true);
    }

    function pickerDeactivate() {
      if (!pickActive) {
        // Still clean up any orphaned DOM (e.g. cancel-select arriving while
        // inactive after a previous teardown left elements somehow).
        if (pickerOverlay && pickerOverlay.parentNode) pickerOverlay.parentNode.removeChild(pickerOverlay);
        if (pickerHint && pickerHint.parentNode) pickerHint.parentNode.removeChild(pickerHint);
        pickerOverlay = null;
        pickerHint = null;
        return;
      }
      pickActive = false;
      document.removeEventListener('mousemove', pickerOnMouseMove, true);
      document.removeEventListener('click', pickerOnClick, true);
      document.removeEventListener('keydown', pickerOnKeyDown, true);
      try { document.documentElement.style.cursor = ''; } catch (e) {}
      if (pickerOverlay && pickerOverlay.parentNode) pickerOverlay.parentNode.removeChild(pickerOverlay);
      if (pickerHint && pickerHint.parentNode) pickerHint.parentNode.removeChild(pickerHint);
      pickerOverlay = null;
      pickerHint = null;
    }

    window.addEventListener('message', function(event) {
      var data = event && event.data;
      if (!data || typeof data !== 'object') return;
      var type = data.type;
      if (type !== 'selector-probe' && type !== 'enter-select' && type !== 'cancel-select') {
        return;
      }
      // Strict origin validation for picker messages. Existing legacy
      // messages in this script use '*' / no origin check; picker is
      // intentionally locked down per R13.
      if (PICKER_ALLOWED_PARENT_ORIGINS.indexOf(event.origin) === -1) return;

      if (type === 'selector-probe') {
        // Parent-initiated readiness probe. Record the origin we're talking
        // to and reply directly to the source frame so we never broadcast.
        pickerLearnedParentOrigin = event.origin;
        if (event.source && event.source.postMessage) {
          try {
            event.source.postMessage({ type: 'selector-ready' }, event.origin);
          } catch (e) {}
        }
        return;
      }
      if (type === 'enter-select') {
        pickerLearnedParentOrigin = event.origin;
        // Ack BEFORE activation so the parent's 500ms ack timer measures
        // iframe responsiveness, not user click speed. Idempotent on
        // already-active: ack again so a duplicate enter-select still
        // satisfies the parent's timer.
        try {
          event.source.postMessage({ type: 'enter-select-ack' }, event.origin);
        } catch (e) {}
        if (pickActive) return;
        pickerActivate();
        return;
      }
      if (type === 'cancel-select') {
        // Idempotent: inactive cancel is a no-op (other than orphan cleanup).
        pickerDeactivate();
        return;
      }
    });

    // ─────────────────────────────────────────────────────────────────────
    // Vite HMR error bridge
    //
    // Vite renders a <vite-error-overlay> custom element inside this iframe
    // whenever a parse/transform error happens in the user's source code.
    // The overlay is visually disjoint from the parent chat composer, so a
    // non-technical user can't easily connect "ugly red box in the preview"
    // to "type into the chat to fix it".
    //
    // Watch for the overlay (added/removed) and forward a sanitized summary
    // to the parent so the chat UI can render a clear "Have AI fix this"
    // banner right above the composer. Reuses pickerLearnedParentOrigin so
    // we never broadcast source-code frames to '*' — the parent always
    // probes for the picker on iframe load (even when the picker is hidden),
    // so this origin is normally set within ms of mount.
    // ─────────────────────────────────────────────────────────────────────
    (function() {
      var VITE_ERROR_TAG = 'VITE-ERROR-OVERLAY';
      var VITE_ERROR_CAPS = {
        message: 1024,
        file: 1024,
        frame: 4 * 1024,
        stack: 8 * 1024,
        path: 512,
      };
      var lastSentSignature = '';
      var scanScheduled = false;

      // Cap a string to a maximum BYTE length (UTF-8). The parent
      // (parseViteErrorMessage) validates inbound payloads with TextEncoder
      // and silently drops the entire message if any field exceeds its cap
      // in bytes. Multibyte content — box-drawing chars in oxc parse-error
      // frames ("╭─[ ... ]"), emoji, non-ASCII source — easily pushes a
      // string under 'cap' code units past 'cap' bytes, which used to drop
      // the banner entirely. Truncate by bytes here so the iframe side
      // agrees with the parent's check; strip trailing U+FFFD from a
      // codepoint split at the boundary.
      function capBytes(s, max) {
        if (!s) return s;
        try {
          var bytes = new TextEncoder().encode(s);
          if (bytes.length <= max) return s;
          var out = new TextDecoder('utf-8').decode(bytes.subarray(0, max));
          return out.replace(/\uFFFD+$/, '');
        } catch (e) {
          // No TextEncoder/TextDecoder: fall back to a conservative slice
          // assuming worst-case 3 bytes per code unit.
          var maxLen = Math.floor(max / 3);
          return s.length > maxLen ? s.slice(0, maxLen) : s;
        }
      }

      function getOverlayPart(root, selector, max) {
        if (!root) return '';
        var el;
        try { el = root.querySelector(selector); } catch (e) { return ''; }
        if (!el) return '';
        var raw = String(el.textContent == null ? '' : el.textContent).trim();
        return capBytes(raw, max);
      }

      function postWithLearnedOrigin(payload, attemptsLeft) {
        // pickerLearnedParentOrigin is set by the picker probe handshake
        // (event.origin from selector-probe). The parent always probes on
        // iframe load, so this is normally set within milliseconds. If we
        // catch the overlay before the probe lands, retry briefly rather
        // than dropping the signal — and never fall back to '*' because the
        // payload contains source-code frames.
        if (pickerLearnedParentOrigin) {
          try {
            window.parent.postMessage(payload, pickerLearnedParentOrigin);
          } catch (e) {}
          return;
        }
        if (attemptsLeft <= 0) return;
        setTimeout(function() {
          postWithLearnedOrigin(payload, attemptsLeft - 1);
        }, 100);
      }

      function findOverlay() {
        try {
          return document.querySelector(VITE_ERROR_TAG.toLowerCase());
        } catch (e) {
          return null;
        }
      }

      // React Router runtime error boundary path. When a component throws
      // during SSR rendering (e.g. "window is not defined" inside a
      // component body that touches window without a useEffect guard),
      // Vite's transform succeeds but the render throws — so no
      // <vite-error-overlay> mounts, and the body has none of the Vite
      // transform-error signatures. React Router catches the throw and
      // renders its default ErrorBoundary:
      //   <main>
      //     <h1>Oops!</h1>
      //     <p>{error.message}</p>
      //     <pre><code>{error.stack}</code></pre>
      //   </main>
      // Match on that DOM shape and pull the project-relative file path
      // out of the first stack frame ("at X (/home/user/website/..:N:N)").
      function detectReactRouterErrorPage() {
        try {
          var h1 = document.querySelector('main h1');
          if (!h1 || (h1.textContent || '').trim() !== 'Oops!') return null;
          var codeEl = document.querySelector('main pre code');
          if (!codeEl) return null;
          var stack = String(codeEl.textContent == null ? '' : codeEl.textContent).trim();
          if (!stack) return null;
          // Stack-shape sanity check: real V8 stacks have "at X (path:N:N)"
          // frames. Rejects false positives where a user's site happens to
          // have a similar DOM shape.
          if (!/\n\s*at\s+[^\n]+\(/.test(stack)) return null;
          var msgEl = document.querySelector('main p');
          var message = String((msgEl && msgEl.textContent) || '').trim();
          if (!message) {
            var firstLine = stack.split('\n')[0] || '';
            var colonIdx = firstLine.indexOf(':');
            message = (colonIdx >= 0 ? firstLine.slice(colonIdx + 1) : firstLine).trim();
          }
          // "at Header (/home/user/website/app/components/Header.tsx:140:14)"
          // Capture the project-relative path + line:col.
          var fileMatch = stack.match(/\/home\/user\/website\/([^\s:)]+):(\d+):(\d+)/);
          var file = '';
          if (fileMatch) {
            file = fileMatch[1] + ':' + fileMatch[2] + ':' + fileMatch[3];
          }
          return {
            type: 'vite-error',
            message: capBytes(message, VITE_ERROR_CAPS.message),
            file: capBytes(file, VITE_ERROR_CAPS.file),
            frame: '',
            stack: capBytes(stack, VITE_ERROR_CAPS.stack),
            path: capBytes(String(window.location.pathname || ''), VITE_ERROR_CAPS.path),
          };
        } catch (e) {
          return null;
        }
      }

      // Plaintext SSR error path. When Vite's SSR transform fails, React
      // Router (and similar Vite SSR setups) returns a 500 HTTP response
      // whose body IS the error — "[vite] Internal Server Error",
      // "Transform failed", "[PARSE_ERROR]" — rendered as plain HTML, NOT
      // as a <vite-error-overlay> custom element. The HMR client never
      // mounts on this page, so the overlay observer alone misses it.
      function detectViteSsrErrorPage() {
        var text = '';
        try {
          text = (document.body && document.body.innerText) || '';
        } catch (e) {
          return null;
        }
        if (!text) return null;
        var hasSignature = (
          text.indexOf('[vite] Internal Server Error') !== -1 ||
          text.indexOf('Transform failed') !== -1 ||
          text.indexOf('[PARSE_ERROR]') !== -1
        );
        if (!hasSignature) return null;
        // Box-drawing line ╭─[ app/routes/_index.tsx:33:18 ] → file path.
        // Escapes are doubled because this whole script is a template literal.
        var fileMatch = text.match(/╭─\[\s*([^\s\]]+)\s*\]/);
        // First "Error: <message>" line → human-readable summary.
        var errMatch = text.match(/Error:\s*([^\n]+)/);
        return {
          type: 'vite-error',
          message: capBytes(
            errMatch ? errMatch[1].trim() : 'Build error in your website code',
            VITE_ERROR_CAPS.message
          ),
          file: capBytes(fileMatch ? fileMatch[1].trim() : '', VITE_ERROR_CAPS.file),
          frame: capBytes(text, VITE_ERROR_CAPS.frame),
          stack: '',
          path: capBytes(String(window.location.pathname || ''), VITE_ERROR_CAPS.path),
        };
      }

      // Auto-recovery poll for the two "stuck page" error types (Vite SSR
      // transform error, React Router runtime ErrorBoundary). Neither
      // ships Vite's HMR client, so even after the underlying file is
      // fixed the iframe sits on a stale error response. Re-fetch the
      // current URL on a bounded backoff; reload when the response body
      // no longer carries an error signature. We deliberately do NOT
      // start the poll for the <vite-error-overlay> case — Vite's own
      // HMR client clears that overlay when the fix arrives, and a
      // reload there would stomp the user's in-page state.
      //
      // All polling code is best-effort and wrapped in try/catch so a
      // bug in this block can never break the rest of the bridge.
      var errorPollState = null;

      function pageStillHasError(html) {
        if (!html) return false;
        if (html.indexOf('[vite] Internal Server Error') !== -1) return true;
        if (html.indexOf('Transform failed') !== -1) return true;
        if (html.indexOf('[PARSE_ERROR]') !== -1) return true;
        // React Router runtime ErrorBoundary signature: the "Oops!" h1
        // co-occurring with "react-dom-server" in the streamed stack.
        // Both strings on their own are too generic; together they
        // only ever co-occur on the RR runtime error page.
        if (html.indexOf('Oops!') !== -1 && html.indexOf('react-dom-server') !== -1) return true;
        return false;
      }

      function startErrorPoll() {
        try {
          if (errorPollState) return;
          var state = {
            startTime: Date.now(),
            nextInterval: 1000,
            timer: null,
            inFlight: false,
            stopped: false,
          };
          var MAX_DURATION_MS = 5 * 60 * 1000;
          var MAX_INTERVAL_MS = 15 * 1000;
          var BACKOFF = 1.5;

          function stop() {
            try {
              state.stopped = true;
              if (state.timer) { clearTimeout(state.timer); state.timer = null; }
            } catch (e) {}
            errorPollState = null;
          }

          function schedule(overrideMs) {
            try {
              if (state.stopped) return;
              var ms = (typeof overrideMs === 'number') ? overrideMs : state.nextInterval;
              state.timer = setTimeout(tick, ms);
              if (typeof overrideMs !== 'number') {
                state.nextInterval = Math.min(MAX_INTERVAL_MS, Math.ceil(state.nextInterval * BACKOFF));
              }
            } catch (e) {}
          }

          function tick() {
            try {
              state.timer = null;
              if (state.stopped) return;
              if (Date.now() - state.startTime > MAX_DURATION_MS) { stop(); return; }
              // Wait until the tab is visible again. Fallback timer in case
              // the visibilitychange handler is missed.
              if (document.hidden) { schedule(MAX_INTERVAL_MS); return; }
              if (state.inFlight) { schedule(); return; }
              state.inFlight = true;
              fetch(window.location.href, { cache: 'no-store', credentials: 'same-origin' })
                .then(function(res) { return res.text(); })
                .then(function(text) {
                  state.inFlight = false;
                  if (state.stopped) return;
                  if (!pageStillHasError(text)) {
                    stop();
                    try { window.location.reload(); } catch (e) {}
                    return;
                  }
                  schedule();
                })
                .catch(function() {
                  state.inFlight = false;
                  schedule();
                });
            } catch (e) {
              state.inFlight = false;
              schedule();
            }
          }

          errorPollState = {
            stop: stop,
            kick: function() {
              try {
                if (state.timer) { clearTimeout(state.timer); state.timer = null; }
                tick();
              } catch (e) {}
            },
          };
          schedule();
        } catch (e) {}
      }

      function stopErrorPoll() {
        try {
          if (errorPollState) errorPollState.stop();
        } catch (e) {}
      }

      document.addEventListener('visibilitychange', function() {
        try {
          if (!document.hidden && errorPollState) errorPollState.kick();
        } catch (e) {}
      });

      function scanAndSend() {
        var overlay = findOverlay();
        if (overlay) {
          var root = null;
          try { root = overlay.shadowRoot; } catch (e) { root = null; }
          if (!root) return;
          var payload = {
            type: 'vite-error',
            message: getOverlayPart(root, '.message-body', VITE_ERROR_CAPS.message)
              || getOverlayPart(root, '.message', VITE_ERROR_CAPS.message),
            file: getOverlayPart(root, '.file', VITE_ERROR_CAPS.file),
            frame: getOverlayPart(root, '.frame', VITE_ERROR_CAPS.frame),
            stack: getOverlayPart(root, '.stack', VITE_ERROR_CAPS.stack),
            path: capBytes(String(window.location.pathname || ''), VITE_ERROR_CAPS.path),
          };
          // Overlay can mount before its shadow DOM is populated; the
          // observer fires again on the populated state, so drop empties.
          if (!payload.message && !payload.file && !payload.frame) return;
          // Dedup: identical errors fire on every HMR ping. Send once per
          // signature; a different error (or cleared+reappeared) gets through.
          var sig = payload.message + '|' + payload.file + '|' + payload.frame;
          if (sig === lastSentSignature) return;
          lastSentSignature = sig;
          postWithLearnedOrigin(payload, 10);
          return;
        }

        // No overlay — check for the plaintext SSR error response page.
        var ssrError = detectViteSsrErrorPage();
        if (ssrError) {
          var ssrSig = ssrError.message + '|' + ssrError.file;
          if (ssrSig === lastSentSignature) return;
          lastSentSignature = ssrSig;
          postWithLearnedOrigin(ssrError, 10);
          startErrorPoll();
          return;
        }

        // …or the React Router runtime error boundary page (component
        // threw during SSR; Vite transform succeeded so neither of the
        // above paths fires).
        var rrError = detectReactRouterErrorPage();
        if (rrError) {
          var rrSig = rrError.message + '|' + rrError.file;
          if (rrSig === lastSentSignature) return;
          lastSentSignature = rrSig;
          postWithLearnedOrigin(rrError, 10);
          startErrorPoll();
          return;
        }

        if (!lastSentSignature) return;
        lastSentSignature = '';
        postWithLearnedOrigin({ type: 'vite-error-cleared' }, 10);
        stopErrorPoll();
      }

      function scheduleScan() {
        if (scanScheduled) return;
        scanScheduled = true;
        setTimeout(function() {
          scanScheduled = false;
          scanAndSend();
        }, 50);
      }

      var viteErrorObserver = new MutationObserver(function(muts) {
        for (var i = 0; i < muts.length; i++) {
          var m = muts[i];
          for (var j = 0; j < m.addedNodes.length; j++) {
            var n = m.addedNodes[j];
            if (n && n.nodeType === 1 && n.tagName === VITE_ERROR_TAG) {
              scheduleScan();
            }
          }
          for (var k = 0; k < m.removedNodes.length; k++) {
            var r = m.removedNodes[k];
            if (r && r.nodeType === 1 && r.tagName === VITE_ERROR_TAG) {
              scheduleScan();
            }
          }
        }
      });
      viteErrorObserver.observe(document.documentElement, {
        childList: true,
        subtree: true,
      });

      // Catch the case where the error is present at script injection time:
      //  - <vite-error-overlay> already mounted (HMR overlay path), OR
      //  - body innerText is the plaintext SSR error response page.
      // Our script runs from <head> before body is parsed, so the SSR-page
      // path needs to wait until the body content is available. The
      // MutationObserver also won't help here because no <vite-error-overlay>
      // is added — the body itself IS the error.
      function initialScan() {
        if (findOverlay() || detectViteSsrErrorPage() || detectReactRouterErrorPage()) {
          scheduleScan();
        }
      }
      if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', initialScan);
      } else {
        initialScan();
      }
    })();
  }
  
})();
